    def get_smooth_curve(self, diameters, percentages, num_points=500):
        """Generate smooth curve using cubic spline interpolation in log space"""
        try:
            x = np.array(diameters, dtype=float)
            y = np.array(percentages, dtype=float)
            
            valid_mask = np.isfinite(x) & np.isfinite(y)
            x = x[valid_mask]
            y = y[valid_mask]
            
            if len(x) < 2:
                return x, y
            
            sort_idx = np.argsort(-x)
            x = x[sort_idx]
            y = y[sort_idx]
            
            unique_x = []
            unique_y = []
            seen = set()
            for i in range(len(x)):
                if x[i] not in seen:
                    seen.add(x[i])
                    unique_x.append(x[i])
                    unique_y.append(y[i])
            
            x = np.array(unique_x)
            y = np.array(unique_y)
            
            if len(x) < 3:
                return x, y
            
            log_x = np.log10(x)
            log_x_fine = np.linspace(log_x.min(), log_x.max(), num_points)
            x_fine = 10 ** log_x_fine
            
            spline = interpolate.CubicSpline(log_x, y, bc_type='natural', extrapolate=False)
            y_fine = spline(log_x_fine)
            
            y_fine = np.maximum.accumulate(y_fine)
            y_fine = np.clip(y_fine, 0, 100)
            
            return x_fine, y_fine
            
        except Exception as e:
            print(f"Smoothing error: {e}")
            return diameters, percentages

    def plot_series_smooth(self, data, label, color, index):
        """Plot a single data series with smoothed cumulative and frequency curves"""
        try:
            if not data or len(data) == 0:
                return
            
            diameters = [d for d, _ in data]
            
            calculator = GranuloCalculator(data, self.app.initial_weight)
            passing = calculator.cumulative_passing
            
            if len(diameters) >= 3:
                smooth_diam, smooth_passing = self.get_smooth_curve(diameters, passing)
                
                self.ax.plot(smooth_diam, smooth_passing, '-', color=color,
                            label=label, linewidth=2.5)
                
                self.ax.plot(diameters, passing, 'o', color=color,
                            markersize=5, markeredgecolor='white', markeredgewidth=1, alpha=0.7)
            else:
                self.ax.plot(diameters, passing, 'o-', color=color,
                            label=label, linewidth=2, markersize=4, markeredgecolor='white')
            
            if self.show_frequency:
                percentages = calculator.percentages
                
                if not hasattr(self, 'freq_ax'):
                    self.freq_ax = self.ax.twinx()
                    self.freq_ax.set_ylabel('Frequency (%)', fontsize=12, color='gray')
                    self.freq_ax.tick_params(axis='y', labelcolor='gray')
                
                if len(diameters) >= 3:
                    sort_idx = np.argsort(-np.array(diameters))
                    diameters_sorted = np.array(diameters)[sort_idx]
                    percentages_sorted = np.array(percentages)[sort_idx]
                    
                    log_diameters = np.log10(diameters_sorted)
                    log_diameters_fine = np.linspace(log_diameters.min(), log_diameters.max(), 200)
                    diameters_fine = 10 ** log_diameters_fine
                    
                    try:
                        spline = interpolate.CubicSpline(log_diameters, percentages_sorted,
                                                        bc_type='natural', extrapolate=False)
                        percentages_smooth = spline(log_diameters_fine)
                        
                        percentages_smooth = np.clip(percentages_smooth, 0, None)
                        
                        self.freq_ax.plot(diameters_fine, percentages_smooth, '--', color=color,
                                         alpha=0.6, linewidth=1.5,
                                         label=f'{label} (freq)')
                    except Exception as e:
                        print(f"Frequency smoothing failed: {e}")
                        self.freq_ax.plot(diameters_sorted, percentages_sorted, '--s', color=color,
                                         alpha=0.5, linewidth=1.5, markersize=3,
                                         label=f'{label} (freq)')
                else:
                    self.freq_ax.plot(diameters, percentages, '--s', color=color,
                                     alpha=0.5, linewidth=1.5, markersize=3,
                                     label=f'{label} (freq)')
                
                all_freqs = []
                if hasattr(self, 'freq_ax'):
                    for line in self.freq_ax.get_lines():
                        if line.get_label().endswith('(freq)'):
                            all_freqs.extend(line.get_ydata())
                    if all_freqs:
                        self.freq_ax.set_ylim(0, max(all_freqs) * 1.2)
                        
        except Exception as e:
            print(f"Error in plot_series_smooth for {label}: {e}")
            import traceback
            traceback.print_exc()
    
    def add_iso_classification(self):

    def update_quick_percentiles(self):
        """Update the quick percentiles display"""
        if not self.data:
            return
        calculator = GranuloCalculator(self.data, self.app.initial_weight)
        params = calculator.get_all_percentiles()
        self.quick_text.delete('1.0', tk.END)
        
        text = " Percentile  |  Diameter (mm)  |  Phi (φ)\n"
        text += "-" * 45 + "\n"
        percentiles = [5, 10, 16, 25, 30, 50, 60, 70, 75, 84, 90, 95, 99.99]
        for p in percentiles:
            mm = params.get(f'p{p}', 0)
            phi = params.get(f'phi{p}', 0)
            if not math.isnan(mm) and mm > 0 and not math.isnan(phi):
                text += f"     {p:2.0f}%     |     {mm:8.4f}    |  {phi:6.3f}\n"
            else:
                text += f"     {p:2.0f}%     |      --      |    --\n"
        
        self.quick_text.insert('1.0', text)
        self.quick_text.config(state=tk.DISABLED)
    
    def toggle_frequency(self):
        """Toggle frequency curve display"""
        self.show_frequency = not self.show_frequency
        if self.show_frequency:
            self.toggle_btn.config(text="📉 Hide Frequency Curves", bg='#e74c3c')
        else:
            self.toggle_btn.config(text="📈 Show Frequency Curves", bg='#3498db')
        if hasattr(self, 'freq_ax'):
            self.freq_ax.remove()
            delattr(self, 'freq_ax')
        self.plot_data()
    
    def on_mouse_move(self, event):
        """Handle mouse movement on graph for cursor tracking"""
        if event.inaxes == self.ax:
            x, y = event.xdata, event.ydata
            if x is not None and y is not None and 0 <= y <= 100:
                self.cursor_x_var.set(f"Diameter: {x:.6f} mm")
                self.cursor_y_var.set(f"Passing: {y:.2f} %")
    
    def on_series_change(self, event=None):
        """Handle series selection change"""
        self.plot_data()
    
    def project_percentile(self, event=None):
        """Project percentile value to diameter"""
        try:
            pct = float(self.proj_entry.get())
            if 0 <= pct <= 100:
                if self.data:
                    calculator = GranuloCalculator(self.data, self.app.initial_weight)
                    diameter = calculator.get_percentile_diameter(pct)
                    if not math.isnan(diameter) and diameter > 0:
                        self.proj_result.config(text=f"→ {diameter:.6f} mm")
                    else:
                        self.proj_result.config(text="→ Outside range")
                else:
                    self.proj_result.config(text="→ No data")
            else:
                self.proj_result.config(text="→ Invalid (0-100)")
        except ValueError:
            self.proj_result.config(text="→ Enter number")

    def show_correlation(self):
        """Show correlation graphs for multiple samples"""

        win.geometry("900x700")
        
        fig = Figure(figsize=(12, 9), dpi=100)

        sample_names = []
        for i, series in enumerate(all_series):

            sample_names.append(f"Sample {i+1}")
        
        skewness = [p.get('skewness', 0) for p in all_params]
        kurtosis = [p.get('kurtosis', 0) for p in all_params]
        sorting = [p.get('sorting', 0) for p in all_params]
        mean_phi = [p.get('mean_phi', 0) for p in all_params]
        mean_mm = [p.get('mean_mm', 0) for p in all_params]
        fm = [p.get('fm', 0) for p in all_params]
        
        ax1 = fig.add_subplot(221)
        scatter1 = ax1.scatter(skewness, kurtosis, c=range(len(skewness)),
                              cmap='viridis', s=100, alpha=0.7)
        ax1.set_xlabel('Skewness (Sk)', fontsize=11)
        ax1.set_ylabel('Kurtosis (KG)', fontsize=11)
        ax1.set_title('Skewness vs Kurtosis\n(Friedman, 1967)', fontsize=12, fontweight='bold')
        ax1.grid(True, alpha=0.3)
        ax1.axhline(y=1.11, color='r', linestyle='--', alpha=0.5, label='Leptokurtic')
        ax1.axhline(y=0.90, color='b', linestyle='--', alpha=0.5, label='Platykurtic')
        ax1.axvline(x=0, color='gray', linestyle='-', alpha=0.3)
        ax1.legend(fontsize=8)

        ax2 = fig.add_subplot(222)
        scatter2 = ax2.scatter(mean_phi, sorting, c=range(len(mean_phi)),
                              cmap='plasma', s=100, alpha=0.7)
        ax2.set_xlabel('Mean Grain Size (φ)', fontsize=11)
        ax2.set_ylabel('Sorting (σI)', fontsize=11)

        ax2.grid(True, alpha=0.3)

        ax3.set_title('Fineness Modulus vs Mean Size', fontsize=12, fontweight='bold')
        ax3.set_xscale('log')
        ax3.grid(True, alpha=0.3)
        
        ax4 = fig.add_subplot(224)
        param_names = ['Skew', 'Kurt', 'Sort', 'Mean φ', 'FM']
        param_matrix = np.array([skewness, kurtosis, sorting, mean_phi, fm])
        corr_matrix = np.corrcoef(param_matrix)
        im = ax4.imshow(corr_matrix, cmap='RdBu', vmin=-1, vmax=1, aspect='auto')
        ax4.set_xticks(range(len(param_names)))
        ax4.set_yticks(range(len(param_names)))

        plt.colorbar(im, ax=ax4, fraction=0.046, pad=0.04)
        fig.tight_layout()

    # ✅ FIX #3: Corrected Passega diagram C and M value calculation
    def show_passega_diagram(self):
        """Show Passega C-M diagram (Passega & Byramjee, 1969)"""
        all_series = []
        if self.data:
            all_series.append(self.data)
        all_series.extend(self.multiple_series)
        
        if len(all_series) < 1:
            messagebox.showwarning("No Data", "Need at least 1 sample")
            return
        
        win = tk.Toplevel(self)
        win.title("Passega C-M Diagram (1969)")
        win.geometry("800x700")
        
        fig = Figure(figsize=(10, 8), dpi=100)
        ax = fig.add_subplot(111)
        
        # ✅ FIX #3: Calculate C (1st percentile) and M (median) for each sample
        c_values = []
        m_values = []
        sample_names = []
        
        for i, series in enumerate(all_series):
            try:
                calc = GranuloCalculator(series, self.app.initial_weight)
                
                # ✅ Get C value (1st percentile - coarsest 1%)
                c = calc.get_percentile_diameter(1)
                
                # ✅ Get M value (50th percentile - median)
                m = calc.get_percentile_diameter(50)
                
                # ✅ Validate C and M values more strictly
                if (c is not None and m is not None and 
                    not math.isnan(c) and not math.isnan(m) and 
                    c > 0 and m > 0 and c < 1000 and m < 1000):
                    
                    c_values.append(c)
                    m_values.append(m)
                    sample_names.append(f"S{i+1}")
                    print(f"Sample {i+1}: C={c:.4f}mm, M={m:.4f}mm")
                else:
                    print(f"Sample {i+1}: Invalid C or M value (C={c}, M={m})")
                    
            except Exception as e:
                print(f"Error calculating C and M for sample {i+1}: {e}")
                continue
        
        # ✅ Better error message with debugging info
        if not c_values or not m_values:
            debug_msg = (
                f"Could not calculate C and M values.\n\n"
                f"Possible causes:\n"
                f"• Insufficient data points (need at least 10 sieves)\n"
                f"• Data range doesn't cover 1st or 50th percentile\n"
                f"• Invalid initial weight: {self.app.initial_weight}g\n"
                f"• Check your sieve data covers 0.001mm to 100mm range\n\n"
                f"Samples processed: {len(all_series)}\n"
                f"Valid C-M pairs: {len(c_values)}"
            )
            messagebox.showwarning("No Valid Data", debug_msg)
            return
        
        # Plot scatter points
        scatter = ax.scatter(m_values, c_values, c=range(len(c_values)),
                            cmap='viridis', s=150, alpha=0.7, edgecolors='black')
        ax.set_xlabel('M - Median (mm)', fontsize=12)
        ax.set_ylabel('C - 1st Percentile (mm)', fontsize=12)
        ax.set_title('Passega C-M Diagram\n(Passega & Byramjee, 1969)',
                    fontsize=14, fontweight='bold')
        
        # Set log scales
        ax.set_xscale('log')
        ax.set_yscale('log')
        
        # Add transportation mechanism fields
        ax.axhspan(0.1, 0.5, xmin=0, xmax=0.3, alpha=0.2, color='blue', label='I - Suspension')
        ax.axhspan(0.5, 10, xmin=0.1, xmax=0.8, alpha=0.2, color='green', label='II - Saltation')
        ax.axhspan(10, 1000, xmin=0.5, xmax=1, alpha=0.2, color='red', label='III - Rolling')
        
        # Add C = M line
        max_val = max(max(c_values), max(m_values))
        min_val = min(min(c_values), min(m_values))
        ax.plot([min_val, max_val], [min_val, max_val], 'k--', alpha=0.5, label='C = M')
        
        # Add sample labels
        for i, name in enumerate(sample_names):
            ax.annotate(name, (m_values[i], c_values[i]),
                       fontsize=10, xytext=(5,5), textcoords='offset points',
                       bbox=dict(boxstyle='round,pad=0.2', facecolor='white', alpha=0.7))
        
        ax.grid(True, alpha=0.3, which='both')
        ax.legend(loc='upper left', fontsize=9)
        
        # Add interpretation
        fig.text(0.5, 0.01,
                "C-M diagram: C = coarsest 1% (competence), M = median (average energy)\n"
                "Fields indicate dominant transportation mechanism (Passega & Byramjee, 1969)",
                ha='center', fontsize=10, style='italic',
                bbox=dict(boxstyle='round', facecolor='lightyellow', alpha=0.8))
        
        fig.tight_layout(rect=[0, 0.03, 1, 0.98])
        
        canvas = FigureCanvasTkAgg(fig, win)
        canvas.draw()
        canvas.get_tk_widget().pack(fill=tk.BOTH, expand=True)
        toolbar = NavigationToolbar2Tk(canvas, win)
        toolbar.update()
